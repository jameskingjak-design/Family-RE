        volunteers: Object.fromEntries(VOLUNTEER_CATEGORIES.map(v => [v.key, []])),

        orders: [],

        shirts: Object.fromEntries(SHIRT_SIZES.map(s => [s, 0]))

      };

    }

    function loadState() {

      try {

        const raw = localStorage.getItem(STORAGE_KEY);

        return raw ? JSON.parse(raw) : defaultState();

      } catch {

        return defaultState();

      }

    }

    function saveState() {

      localStorage.setItem(STORAGE_KEY, JSON.stringify(state));

    }

    function uid() {

      return Math.random().toString(16).slice(2) + Date.now().toString(16);

    }

    function initials(name) {

      return String(name || "")

        .trim()

        .split(/\s+/)

        .slice(0, 2)

        .map(part => part[0] ? part[0].toUpperCase() : "")

        .join("") || "OF";

    }

    function currency(n) {

      return new Intl.NumberFormat("en-US", { style: "currency", currency: "USD" }).format(Number(n || 0));

    }

    function escapeHtml(str) {

      return String(str || "")

        .replaceAll("&", "&amp;")

        .replaceAll("<", "&lt;")

        .replaceAll(">", "&gt;")

        .replaceAll('"', "&quot;")

        .replaceAll("'", "&#039;");

    }

    const $ = (id) => document.getElementById(id);

    let state = loadState();

    $("year").textContent = new Date().getFullYear();

    $("payLink").href = PAYMENT_LINK;

    function setSelectOptions(select, values) {

      select.innerHTML = values.map(v => `<option value="${escapeHtml(v)}">${escapeHtml(v)}</option>`).join("");

    }

    function renderStats() {

      const totalMembers = BRANCHES.reduce((sum, branch) => sum + state.branches[branch].length, 0);

      const totalVolunteers = Object.values(state.volunteers).reduce((sum, arr) => sum + arr.length, 0);

      const totalPaidPeople = state.orders.reduce((sum, order) => sum + Number(order.people || 0), 0);

      const totalPaid = totalPaidPeople * 25;

      $("statMembers").textContent = totalMembers;

      $("statVolunteers").textContent = totalVolunteers;

      $("statPaidPeople").textContent = totalPaidPeople;

      $("statTotalPaid").textContent = currency(totalPaid);

    }

    function renderStatement() {

      $("familyStatement").value = state.statement;

    }

    function renderTreeVisual() {

      const wrap = $("treeVisualChildren");

      wrap.innerHTML = BRANCHES.map(branch => {

        const count = state.branches[branch].length;

        return `

          <div class="person-box">

            <h3 style="font-size:16px">${escapeHtml(branch)}</h3>

            <p>${count} signed up</p>

          </div>

        `;

      }).join("");

    }

    function renderTreeList() {

      const search = $("treeSearch").value.trim().toLowerCase();

      const list = $("treeList");

      const branches = BRANCHES.filter(branch => {

        if (!search) return true;

        const names = state.branches[branch].map(person => person.name.toLowerCase()).join(" ");

        return branch.toLowerCase().includes(search) || names.includes(search);

      });

      list.innerHTML = branches.map(branch => {

        const members = state.branches[branch];

        return `

          <details class="branch">

            <summary>

              <span>${escapeHtml(branch)}</span>

              <span class="pill">${members.length} signed up</span>

            </summary>

            <div class="branch-body">

              <div class="member-list">

                ${

                  members.length

                    ? members.map(member => `

                      <span class="member-chip">

                        <span class="avatar">${initials(member.name)}</span>

                        <span>${escapeHtml(member.name)}</span>

                      </span>

                    `).join("")

                    : `<span class="small">No one has signed up under this branch yet.</span>`

                }

              </div>

              <button class="btn" type="button" onclick="openSignup('${escapeHtml(branch)}')">

                Add someone to ${escapeHtml(branch)}

              </button>

            </div>

          </details>

        `;

      }).join("");

    }

    function renderVolunteerForm() {

      const categorySelect = $("volCategory");

      categorySelect.innerHTML = VOLUNTEER_CATEGORIES.map(cat =>

        `<option value="${cat.key}">${cat.title}</option>`

      ).join("");

      if (!categorySelect.value) categorySelect.value = VOLUNTEER_CATEGORIES[0].key;

      updateVolunteerTasks();

    }

    function updateVolunteerTasks() {

      const key = $("volCategory").value;

      const category = VOLUNTEER_CATEGORIES.find(c => c.key === key) || VOLUNTEER_CATEGORIES[0];

      $("volCategoryDesc").textContent = category.description;

      $("volTasks").innerHTML = category.tasks.map(task => `

        <label style="display:flex;align-items:center;gap:8px;margin-bottom:8px;font-weight:400">

          <input type="checkbox" value="${escapeHtml(task)}" />

          <span>${escapeHtml(task)}</span>

        </label>

      `).join("");

    }

    function renderVolunteerRosters() {

      const target = $("volunteerRosters");

      target.innerHTML = VOLUNTEER_CATEGORIES.map(cat => {

        const items = state.volunteers[cat.key];

        return `

          <details class="branch" ${items.length ? "" : ""}>

            <summary>

              <span>${cat.title}</span>

              <span class="pill">${items.length} volunteer${items.length === 1 ? "" : "s"}</span>

            </summary>

            <div class="branch-body">

              ${

                items.length

                  ? items.map(item => `

                    <div class="vol-category" style="margin-bottom:10px">

                      <strong>${escapeHtml(item.name)}</strong>

                      <div class="small">${escapeHtml(item.email || "")}${item.email && item.phone ? " • " : ""}${escapeHtml(item.phone || "")}</div>

                      ${item.availability ? `<div class="small" style="margin-top:6px"><strong>Availability:</strong> ${escapeHtml(item.availability)}</div>` : ""}

                      ${item.tasks.length ? `<div style="margin-top:8px">${item.tasks.map(t => `<span class="pill" style="margin:3px 6px 0 0">${escapeHtml(t)}</span>`).join("")}</div>` : ""}

                    </div>

                  `).join("")

                  : `<span class="small">No volunteers yet in this category.</span>`

              }

            </div>

          </details>

        `;

      }).join("");

    }

    function renderShirtGrid() {

      $("shirtGrid").innerHTML = SHIRT_SIZES.map(size => `

        <div class="shirt-item">

          <strong>${size}</strong>

          <input type="number" min="0" value="${state.shirts[size] || 0}" style="width:84px" data-shirt-size="${size}" />

        </div>

      `).join("");

      document.querySelectorAll("[data-shirt-size]").forEach(input => {

        input.addEventListener("input", () => {

          const size = input.getAttribute("data-shirt-size");

          state.shirts[size] = Number(input.value || 0);

          saveState();

        });

      });

    }

    function updateDuesTotal() {

      const people = Math.max(0, Number($("orderPeople").value || 0));

      $("duesTotal").textContent = currency(people * 25);

      $("duesBreakdown").textContent = `$25 × ${people} ${people === 1 ? "person" : "people"}`;

    }

    function orderCardHtml(order) {

      const shirts = Object.entries(order.shirts || {})

        .map(([size, qty]) => `<span class="pill" style="margin:3px 6px 0 0">${size}: ${qty}</span>`)

        .join("");

      return `

        <div class="order-card" style="margin-bottom:12px">

          <div style="display:flex;justify-content:space-between;gap:12px;flex-wrap:wrap">

            <div>

              <strong style="font-size:16px">${escapeHtml(order.name)}</strong>

              <div class="small">${escapeHtml(order.email || "")}</div>

              <div class="small" style="margin-top:6px">

                <strong>Branch:</strong> ${escapeHtml(order.branch)} |

                <strong>People:</strong> ${order.people} |

                <strong>Dues:</strong> ${currency(order.duesTotal)}

              </div>

              ${shirts ? `<div style="margin-top:8px">${shirts}</div>` : ""}

              ${order.notes ? `<div class="small" style="margin-top:8px"><strong>Notes:</strong> ${escapeHtml(order.notes)}</div>` : ""}

            </div>

            <div>

              <span class="pill">${order.paymentStatus === "paid" ? "Paid" : "Unpaid"}</span>

            </div>

          </div>

        </div>

      `;

    }

    function renderOrders() {

      const preview = $("ordersListPreview");

      const full = $("ordersListFull");

      if (!state.orders.length) {

        preview.innerHTML = `<div class="order-card"><span class="small">No orders yet.</span></div>`;

        full.innerHTML = `<div class="order-card"><span class="small">No orders yet.</span></div>`;

        return;

      }

      preview.innerHTML = `

        <h3 style="margin-top:0">Recent Orders</h3>

        ${state.orders.slice(0, 3).map(orderCardHtml).join("")}

      `;

      full.innerHTML = `

        <h3 style="margin-top:0">All Orders</h3>

        ${state.orders.map(orderCardHtml).join("")}

      `;

    }

    function renderAll() {

      renderStats();

      renderStatement();

      renderTreeVisual();

      renderTreeList();

      renderVolunteerForm();

      renderVolunteerRosters();

      renderShirtGrid();

      renderOrders();

      updateDuesTotal();

    }

    function openSignup(branchName = BRANCHES[0]) {

      $("signupBranch").value = branchName;

      $("signupName").value = "";

      $("signupEmail").value = "";

      $("signupNotes").value = "";

      $("signupDialog").showModal();

    }

    $("openSignupBtn").addEventListener("click", () => openSignup());

    $("quickAddBtn").addEventListener("click", () => openSignup($("quickBranch").value));

    $("closeSignupBtn").addEventListener("click", () => $("signupDialog").close());

    $("submitSignupBtn").addEventListener("click", () => {

      const branch = $("signupBranch").value;

      const name = $("signupName").value.trim();

      const email = $("signupEmail").value.trim();

      const notes = $("signupNotes").value.trim();

      if (!name) {

        alert("Please enter a name.");

        return;

      }

      state.branches[branch].unshift({

        id: uid(),

        name,

        email,

        notes,

        createdAt: new Date().toISOString()

      });

      saveState();

      renderAll();

      $("signupDialog").close();

    });

    $("familyStatement").addEventListener("input", (e) => {

      state.statement = e.target.value;

      saveState();

    });

    $("treeSearch").addEventListener("input", renderTreeList);

    $("volCategory").addEventListener("change", updateVolunteerTasks);

    $("addVolunteerBtn").addEventListener("click", () => {

      const key = $("volCategory").value;

      const name = $("volName").value.trim();

      const email = $("volEmail").value.trim();

      const phone = $("volPhone").value.trim();

      const availability = $("volAvailability").value.trim();

      const tasks = Array.from($("#volTasks").querySelectorAll("input[type='checkbox']:checked")).map(cb => cb.value);

      if (!name) {

        alert("Please enter a name.");

        return;

      }

      state.volunteers[key].unshift({

        id: uid(),

        name,

        email,

        phone,

        availability,

        tasks,

        createdAt: new Date().toISOString()

      });

      $("volName").value = "";

      $("volEmail").value = "";

      $("volPhone").value = "";

      $("volAvailability").value = "";

      updateVolunteerTasks();

      saveState();

      renderAll();

    });

    $("orderPeople").addEventListener("input", updateDuesTotal);

    $("submitOrderBtn").addEventListener("click", () => {

      const name = $("orderName").value.trim();

      const email = $("orderEmail").value.trim();

      const branch = $("orderBranch").value;

      const people = Math.max(0, Number($("orderPeople").value || 0));

      const notes = $("orderNotes").value.trim();

      if (!name) {

        alert("Please enter a name.");

        return;

      }

      const shirts = Object.fromEntries(

        Object.entries(state.shirts).filter(([, qty]) => Number(qty) > 0)

      );

      state.orders.unshift({

        id: uid(),

        name,

        email,

        branch,

        people,

        duesTotal: people * 25,

        shirts,

        notes,

        paymentStatus: "unpaid",

        createdAt: new Date().toISOString()

      });

      $("orderName").value = "";

      $("orderEmail").value = "";

      $("orderPeople").value = 1;

      $("orderNotes").value = "";

      state.shirts = Object.fromEntries(SHIRT_SIZES.map(s => [s, 0]));

      saveState();

      renderAll();

    });

      tab.addEventListener("click", () => {

        document.querySelectorAll(".tab").forEach(t => t.classList.remove("active"));

        tab.classList.add("active");

        const target = tab.getAttribute("data-tab");

        ["pay","orders","payment-help"].forEach(id => {

          $("tab-" + id).classList.toggle("hidden", id !== target);

        });

      });

    });

    $("exportBtn").addEventListener("click", () => {

      const blob = new Blob([JSON.stringify(state, null, 2)], { type: "application/json" });

      const url = URL.createObjectURL(blob);

      const a = document.createElement("a");

      a.href = url;

      a.download = "owens-family-reunion-data.json";

      a.click();

      URL.revokeObjectURL(url);

    });

    setSelectOptions($("quickBranch"), BRANCHES);

    setSelectOptions($("signupBranch"), BRANCHES);

    setSelectOptions($("orderBranch"), BRANCHES);

    renderAll();
